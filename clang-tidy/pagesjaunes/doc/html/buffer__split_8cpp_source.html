288  void
289  BufferSplitTest::SHA256::transform(const unsigned char *message, unsigned int block_nb)
290  {
291  uint32 w[64];
292  uint32 wv[8];
293  uint32 t1, t2;
294  const unsigned char *sub_block;
295  int i;
296  int j;
297  for (i = 0; i < (int) block_nb; i++) {
298  sub_block = message + (i << 6);
299  for (j = 0; j < 16; j++) {
300  SHA2_PACK32(&sub_block[j << 2], &w[j]);
301  }
302  for (j = 16; j < 64; j++) {
303  w[j] = SHA256_F4(w[j - 2]) + w[j - 7] + SHA256_F3(w[j - 15]) + w[j - 16];
304  }
305  for (j = 0; j < 8; j++) {
306  wv[j] = m_h[j];
307  }
308  for (j = 0; j < 64; j++) {
309  t1 = wv[7] + SHA256_F2(wv[4]) + SHA2_CH(wv[4], wv[5], wv[6])
310  + sha256_k[j] + w[j];
311  t2 = SHA256_F1(wv[0]) + SHA2_MAJ(wv[0], wv[1], wv[2]);
312  wv[7] = wv[6];
313  wv[6] = wv[5];
314  wv[5] = wv[4];
315  wv[4] = wv[3] + t1;
316  wv[3] = wv[2];
317  wv[2] = wv[1];
318  wv[1] = wv[0];
319  wv[0] = t1 + t2;
320  }
321  for (j = 0; j < 8; j++) {
322  m_h[j] += wv[j];
323  }
324  }

342  void
343  BufferSplitTest::SHA256::update(const unsigned char *message, unsigned int len)
344  {
345  unsigned int block_nb;
346  unsigned int new_len, rem_len, tmp_len;
347  const unsigned char *shifted_message;
348  tmp_len = BufferSplitTest::SHA256::SHA224_256_BLOCK_SIZE - m_len;
349  rem_len = len < tmp_len ? len : tmp_len;
350  memcpy(&m_block[m_len], message, rem_len);
351  if (m_len + len < BufferSplitTest::SHA256::SHA224_256_BLOCK_SIZE) {
352  m_len += len;
353  return;
354  }
355  new_len = len - rem_len;
356  block_nb = new_len / BufferSplitTest::SHA256::SHA224_256_BLOCK_SIZE;
357  shifted_message = message + rem_len;
358  transform(m_block, 1);
359  transform(shifted_message, block_nb);
360  rem_len = new_len % BufferSplitTest::SHA256::SHA224_256_BLOCK_SIZE;
361  memcpy(m_block, &shifted_message[block_nb << 6], rem_len);
362  m_len = rem_len;
363  m_tot_len += (block_nb + 1) << 6;
364  }

366  void
367  BufferSplitTest::SHA256::final(unsigned char *digest)
368  {
369  unsigned int block_nb;
370  unsigned int pm_len;
371  unsigned int len_b;
372  int i;
373 
374  block_nb = (1 + ((BufferSplitTest::SHA256::SHA224_256_BLOCK_SIZE - 9)
375  < (m_len % BufferSplitTest::SHA256::SHA224_256_BLOCK_SIZE)));
376  len_b = (m_tot_len + m_len) << 3;
377  pm_len = block_nb << 6;
378  memset(m_block + m_len, 0, pm_len - m_len);
379  m_block[m_len] = 0x80;
380  SHA2_UNPACK32(len_b, m_block + pm_len - 4);
381  transform(m_block, block_nb);
382  for (i = 0 ; i < 8; i++)
383  SHA2_UNPACK32(m_h[i], &digest[i << 2]);
384  }

386  std::string
387  BufferSplitTest::sha256(const std::string& input)
388  {
389  unsigned char digest[BufferSplitTest::SHA256::DIGEST_SIZE];
390  memset(digest, 0, BufferSplitTest::SHA256::DIGEST_SIZE);
391 
392  BufferSplitTest::SHA256 ctx = BufferSplitTest::SHA256();
393  ctx.init();
394  ctx.update(reinterpret_cast<const unsigned char *>(input.c_str()), input.length());
395  ctx.final(digest);

398  buf[2 * BufferSplitTest::SHA256::DIGEST_SIZE] = 0;

400  sprintf(buf+i*2, "%02x", digest[i]);
