59  void
60  MacroExpands(const Token &macroNameTok,
61  const MacroDefinition &md, SourceRange range,
62  const MacroArgs *args) override
63  {
64  FileID main_fid = m_src_mgr.getMainFileID();
65  SourceLocation defloc = md.getMacroInfo()->getDefinitionLoc();
66  StringRef macroName = macroNameTok.getIdentifierInfo()->getName();
67  std::string macroLiteral;
68  FileID name_fid = m_src_mgr.getFileID(defloc);
69 
70  if (name_fid == main_fid)
71  {
72  SourceRange sr;
73  bool have_literal = false;
74  for (const auto& t : md.getMacroInfo()->tokens())
75  {
76  if (t.is(tok::string_literal))
77  {
78  SourceLocation sloc = t.getLocation();
79  SourceLocation eloc = t.getEndLoc();
80  sr = SourceRange(sloc, eloc);
81  have_literal = true;
82  }
83  else if (t.is(tok::wide_string_literal) ||
84  t.is(tok::angle_string_literal) ||
85  t.is(tok::utf8_string_literal) ||
86  t.is(tok::utf16_string_literal) ||
87  t.is(tok::utf32_string_literal))
88  {
89  // TODO: handle that kind of string charsets
90  std::string slKind;
91  if (t.is(tok::wide_string_literal))
92  slKind = "Wide String";
93  else if (t.is(tok::angle_string_literal))
94  slKind = "Angle String";
95  else if (t.is(tok::utf8_string_literal))
96  slKind = "UTF8 String";
97  else if (t.is(tok::utf16_string_literal))
98  slKind = "UTF16 String";
99  else if (t.is(tok::utf32_string_literal))
100  slKind = "UTF32 String";
101 
102  //errs() << "*** Token for weird string (" << slKind << ") found\n";
103  m_parent->emitError(m_diag_engine,
104  t.getLocation(),
105  ExecSQLOpenToFunctionCall::EXEC_SQL_2_FUNC_ERROR_UNSUPPORTED_STRING_CHARSET,
106  &slKind);
107  }
108  }
109 
110  if (have_literal)
111  {
112  //outs() << "Adding macro '" << macroName << "' expansion at " << range.getBegin().printToString(m_src_mgr) << "...'\n";
113  ExecSQLOpenToFunctionCall::SourceRangeForStringLiterals *ent
114  = new ExecSQLOpenToFunctionCall::SourceRangeForStringLiterals(range, sr, macroName);
115  m_macrosStringLiteralsPtr->insert(ent);
116  }
117  }

367  /*
368  * ExecSQLOpenToFunctionCall::emitDiagAndFix
369  *
370  * @brief Emit a diagnostic message and possible replacement fix for each
371  * statement we will be notified with.
372  *
373  * This method is called each time a statement to handle (rewrite) is found.
374  * One replacement will be emited for eachnode found.
375  * It is passed all necessary arguments for:
376  * - creating a comprehensive diagnostic message
377  * - computing the locations of code we will replace
378  * - computing the new code that will replace old one
379  *
380  * @param loc_start The CompoundStmt start location
381  * @param loc_end The CompoundStmt end location
382  * @param function_name The function name that will be called
383  */
384  std::string
385  ExecSQLOpenToFunctionCall::emitDiagAndFix(const SourceLocation& loc_start,
386  const SourceLocation& loc_end,
387  const std::string& function_name,
388  const std::string& function_args)
389  {
390  /* Range of the statement to change */
391  SourceRange stmt_range(loc_start, loc_end);
392 
393  /* Diagnostic builder for a found AST node */
394  /* Default is a warning, and it is emitted */
395  /* as soon as the diag builder is destroyed */
396  DiagnosticBuilder mydiag = diag(loc_end,
397  "ProC Statement Block shall be replaced"
398  " by a function call named '%0'")
399  << function_name;
400 
401  /* Replacement code built */
402  std::string replt_code = function_name;
403  replt_code.append(std::string("("));
404  replt_code.append(function_args);
405  replt_code.append(std::string(");"));
406 
407  /* Emit the replacement over the found statement range */
408  mydiag << FixItHint::CreateReplacement(stmt_range, replt_code);
409 
410  return replt_code;
411  }

426  bool
427  ExecSQLOpenToFunctionCall::processTemplate(const std::string& tmpl,
428  const std::string& fname,
429  string2_map& values_map)
430  {
431  // Return value
432  bool ret = false;
433  // File buffers used for I/O
434  std::filebuf fbi, fbo;
435 
436  do
437  {
438  // Open template
439  if (!fbi.open(tmpl, std::ios::in))
440  break;
441 
442  // Open output file
443  if (!fbo.open(fname, std::ios::out))
444  break;
445 
446  // Input and output streams
447  std::istream is(&fbi);
448  std::ostream os(&fbo);
449 
450  // While not end of template
451  while (is)
452  {
453  // Buffer for input line
454  char buf[1024];
455  // Read line in buffer
456  is.getline(buf, 255);
457  std::string aline = buf;
458  // Iterate on all map values
459  for (auto it = values_map.begin();
460  it != values_map.end();
461  ++it)
462  {
463  // Position of found value name
464  size_t pos = 0;
465  do
466  {
467  // If we found the current value name
468  if ((pos = aline.find(it->first, pos)) != std::string::npos)
469  {
470  // Erase value name at found position in input line
471  aline.erase(pos, it->first.length());
472  // Insert value replacement at the same position
473  aline.insert(pos, it->second);
474  }
475  }
476  // Do it until end of line
477  while (pos != std::string::npos);
478  }
479 
480  // Output processed line
481  os << aline << std::endl;
482  }
483 
484  // All was ok
485  ret = true;
486  }
487  while (0);
488 
489  // Close output file
490  if (fbo.is_open())
491  fbo.close();
492  // Close input file
493  if (fbi.is_open())
494  fbi.close();
495 
496  // Return status
497  return ret;
498  }

524  std::string fileName = values_map["@RequestFunctionName@"];

530 
531  // Handling dir creation errors

534  dirName.append("/");
535  fileName.insert(0, "/");
536  fileName.insert(0, dirName);
537  fileName.append(GENERATION_SOURCE_FILENAME_EXTENSION);
538 
539  if (mkdret == -1 && errno != EEXIST)
540  emitError(diag_engine,
541  dummy,
542  ExecSQLOpenToFunctionCall::EXEC_SQL_2_FUNC_ERROR_SOURCE_CREATE_DIR,

544  else if (!generate_req_allow_overwrite &&
545  (stat (fileName.c_str(), &buffer) == 0))
546  emitError(diag_engine,
547  dummy,
548  ExecSQLOpenToFunctionCall::EXEC_SQL_2_FUNC_ERROR_SOURCE_EXISTS,

1060  void
1061  ExecSQLOpenToFunctionCall::check(const MatchFinder::MatchResult &result)
1062  {
1063  outs() << "void ExecSQLOpenToFunctionCall::check(const MatchFinder::MatchResult &result): ENTRY!\n";
1064  map_replacement_values rv;
1065 
1066  // Get the source manager
1067  SourceManager &srcMgr = result.Context->getSourceManager();
1068  DiagnosticsEngine &diagEngine = result.Context->getDiagnostics();
1069 
1070  /*
1071  * Init check context
1072  * ------------------
1073  */
1074 
1075  // Get the compound statement AST node as the bounded var 'proCBlock'
1076  const CompoundStmt *stmt = result.Nodes.getNodeAs<CompoundStmt>("proCBlock");
1077  const FunctionDecl *curFunc = result.Nodes.getNodeAs<FunctionDecl>("function");
1078 
1079  // Get start/end locations for the statement
1080  SourceLocation loc_start = stmt->getLocStart();
1081  SourceLocation loc_end = stmt->getLocEnd();
1082 
1083  // Get file ID in src mgr
1084  FileID startFid = srcMgr.getFileID(loc_start);
1085  // Get compound statement start line num
1086  unsigned startLineNum = srcMgr.getLineNumber(startFid, srcMgr.getFileOffset(loc_start));
1087  std::stringbuf slnbuffer;
1088  std::ostream slnos (&slnbuffer);
1089  slnos << startLineNum;
1090  std::string originalSourceFileBasename = srcMgr.getFileEntryForID(srcMgr.getMainFileID())->getName().str();
1091  // If original file name is a path, keep only basename
1092  // (erase all before / and all after .)
1093  if (originalSourceFileBasename.rfind("/") != std::string::npos)
1094  originalSourceFileBasename.erase(0, originalSourceFileBasename.rfind("/")+1);
1095  originalSourceFileBasename.erase(originalSourceFileBasename.rfind("."), std::string::npos);
1096  std::string originalSourceFilename = srcMgr.getFileEntryForID(srcMgr.getMainFileID())->getName().str().append(slnbuffer.str().insert(0, "#"));
1097 
1098  //outs() << "Found one result at line " << startLineNum << " of file '" << originalSourceFilename << "\n";
1099 
1100  /*
1101  * Find the comment for the EXEC SQL statement
1102  * -------------------------------------------
1103  *
1104  * Iterate from line +2 until finding the whole EXEC SQL comment
1105  * Line +2 because line+1, when line numbers are generated by ProC, contains start of EXEC SQL
1106  */
1107 
1108  // Current line number
1109  size_t lineNum = startLineNum+2;
1110 
1111  // Get end comment loc
1112  SourceLocation comment_loc_end = srcMgr.translateLineCol(startFid, lineNum, 1);
1113  SourceLocation comment_loc_start;
1114 
1115  // Does error occured while doing getCharacterData
1116  bool errOccured = false;
1117 
1118  // Get comment C string from the file memory buffer
1119  const char *commentCStr = srcMgr.getCharacterData(comment_loc_end, &errOccured);
1120 
1121  // The line data & comment
1122  std::string lineData;
1123 
1124  // Offset of comment start in file
1125  size_t commentStart = 0;
1126 
1127  // If #line is available keep line number and filename of the .pc file
1128  unsigned int pcLineNumStart = 0;
1129  unsigned int pcLineNumEnd = 0;
1130  std::string pcFilename;
1131  bool found_line_info = false;
1132 
1133  /*
1134  * Finding loop
1135  * ------------
1136  */
1137  do
1138  {
1139  // All was ok ?
1140  if (!errOccured)
1141  {
1142  // Assign comment C string to line data var
1143  lineData.assign(commentCStr);
1144  // And remove all remaining chars after end of line
1145  lineData.erase (lineData.find("\n", 1), std::string::npos);
1146 
1147  // Try to get comment start in current line
1148  commentStart = lineData.find("/*");
1149 
1150  if (lineData.find("#line ") != std::string::npos)
1151  {
1152  // Keep the line number and file:
1153  // Format of the preprocessor #line
1154  // #line <linenum> <filepath>
1155  // linenum is an unsigned int
1156  // and filepath is a dbl-quoted string
1157  Regex lineDefineRe(PAGESJAUNES_REGEX_EXEC_SQL_ALL_LINE_DEFINE_RE);
1158  SmallVector<StringRef, 8> lineMatches;
1159  if (lineDefineRe.match(lineData, &lineMatches))
1160  {
1161  found_line_info = true;
1162  std::istringstream isstr;
1163  std::stringbuf *pbuf = isstr.rdbuf();
1164  pbuf->str(lineMatches[1]);
1165  if (pcLineNumStart)
1166  isstr >> pcLineNumEnd;
1167  else
1168  isstr >> pcLineNumStart;
1169  pcFilename = lineMatches[2];
1170  }
1171  else
1172  {
1173  errs() << "Cannot match a #line definition !\n";
1174  }
1175  }
1176 
1177  // If found, break the loop
1178  if (commentStart != std::string::npos)
1179  break;
1180  }
1181 
1182  else
1183  // Error occured at getting char data ! break comment start find loop
1184  break;
1185 
1186  // try previous line
1187  lineNum--;
1188  // Get Location for the new current line
1189  comment_loc_start = srcMgr.translateLineCol(startFid, lineNum, 1);
1190  // And get char data again
1191  commentCStr = srcMgr.getCharacterData(comment_loc_start, &errOccured);
1192  }
1193  // Try again
1194  while (lineNum > 0);
1195 
1196  /*
1197  * Comment found
1198  */
1199  std::string comment;
1200 
1201  // If found comment start & no error occured
1202  if (lineNum > 0 && !errOccured)
1203  {
1204  // Assign comment char data in the comnt string
1205  comment = std::string(commentCStr);
1206 
1207  /*
1208  * Erase from end of comment to end of data
1209  */
1210 
1211  // Find end of comment
1212  size_t endCommentPos = comment.find("*/", 1);
1213 
1214  // Erase until end of character data
1215  comment.erase (endCommentPos +2, std::string::npos);
1216 
1217  /*
1218  * Remove all \n from comment
1219  */
1220

1222  size_t crpos = comment.find("\n", 0);
1223  bool had_cr = false;
1224 
1225  // Iterate
1226  do
1227  {
1228  // If one has been found
1229  if (crpos != std::string::npos)
1230  {
1231  // Erase CR
1232  comment.erase(crpos, 1);
1233  had_cr = true;
1234  }
1235  // Find again CR
1236  crpos = comment.find("\n", 0);
1237  }
1238  // Until no more is found
1239  while (crpos != std::string::npos);
1240 
1241  outs() << "comment found for compound statement at line #" << startLineNum << ": '" << comment << "'\n";

1244  * Create function call for the request
1245  */

1249

1254  * - RequestFunctionName
1255  * - RequestFunctionArgs

1258  * - RequestExecSql
1259  */

1261  std::string requestFunctionArgs;
1262  std::string requestExecSqlDeclareSection;

1267

1272  * Now we match against a more permissive regex for all other (simpler) requests
1273  * =============================================================================
1274  */
1275  if (openReqRe.match(comment, &matches))
1276  {
1277  /*
1278  * Find the request var assignment
1279  */
1280 
1281  // The request name used in ProC
1282  std::string reqName = matches[PAGESJAUNES_REGEX_EXEC_SQL_OPEN_REQ_RE_REQNAME];

1292  if (!hvm["hostvar"].empty())
1293  {
1294  // Find the corresponding declaration in the function
1295  string2_map hostVarMap = findDeclInFunction(curFunc, hvm["hostrecord"]);
1296  // If the host variable is a field of a struct
1297  if (!hvm["deref"].empty())
1298  {
1299  // Find the struct definition for getting the field
1300  string2_map memberVarMap = findCXXRecordMemberInTranslationUnit(result.Context->getTranslationUnitDecl(),
1301  hostVarMap["typeName"],
1302  hvm["hostmember"]);
1303  // Let's rename the parameter by prepending 'a_'
1304  // FIXME: handle potential collisions on param names

1311  memberVarMap["elementSize"],

1315  memberVarMap["elementSize"],
1316  memberVarMap["fieldName"],
1317  paramName));

1319  memberVarMap["elementType"],
1320  memberVarMap["elementSize"]));
1321  requestFunctionArgs.append(createParamsCall(hvm["hostvar"]));
1322  newHostVarList.append(createHostVarList(memberVarMap["fieldName"]));
1323  }
1324  // Host var is a standard decl in the function
1325  else
1326  {
1327  // Let's rename the parameter by prepending 'a_'
1328  std::string paramName = "a_";
1329  paramName.append(hostVarMap["symName"]);
1330 
1331  // Format the various usage we make of it
1332  requestFunctionParamsDef.append(createParamsDef(hostVarMap["typeName"],
1333  hostVarMap["elementType"],
1334  hostVarMap["elementSize"],

1338  hostVarMap["elementSize"],
1339  hostVarMap["symName"],
1340  paramName));
1341  requestFunctionParamsDecl.append(createParamsDecl(hostVarMap["typeName"],
1342  hostVarMap["elementType"],
1343  hostVarMap["elementSize"]));
1344  requestFunctionArgs.append(createParamsCall(hostVarMap["symName"]));
1345  newHostVarList.append(createHostVarList(hostVarMap["symName"]));
1346  }
1347  }

1350  if (!hvm["hostvari"].empty())
1351  {
1352  // Find the decl in func
1353  string2_map hostIndicVarMap = findDeclInFunction(curFunc, hvm["hostrecordi"]);
1354  // Member of a struct

1357  // Find the struct definition for getting field specification

1368  memberIndicVarMap["elementSize"],

1373  memberIndicVarMap["fieldName"],

1414
