313  void
314  DeIncludePreProC::check(const MatchFinder::MatchResult &result)
315  {
316  // Comment regular expression for a processed header
317  Regex comment_re(comment_regex);
318  // Translation unit found by matcher
319  const TranslationUnitDecl *translation_unit =
320  result.Nodes.getNodeAs<TranslationUnitDecl>("translation_unit");
321 
322  // Found a translation unit
323  if (translation_unit)
324  {
325  // AST context
326  ASTContext &ast_ctxt = translation_unit->getASTContext();
327  // Raw comments list
328  RawCommentList &raw_comments = ast_ctxt.getRawCommentList();
329  // Get an array of the comments
330  ArrayRef<RawComment *> raw_comments_array = raw_comments.getComments();
331 
332  // Filter comments for finding all includes
333  for (auto cit = raw_comments_array.begin();
334  cit != raw_comments_array.end();
335  ++cit)
336  {
337  // Matches result
338  SmallVector<StringRef, 8> matches;
339  // Raw text for the comment
340  std::string raw_text = (*cit)->getRawText(ast_ctxt.getSourceManager()).str();
341  // Raw comment
342  RawComment *raw_comment = (*cit);
343  // Location of the comment start
344  SourceLocation comment_start_loc = raw_comment->getLocStart();
345 
346  // If comment matches
347  if (comment_re.match(raw_text, &matches))
348  {
349  // Header file name
350  std::string header_name = matches[1].str();
351 
352  if ((toIncludeIn.empty() ||
353  contain(toIncludeIn, header_name)) &&
354  (toExcludeFrom.empty() ||
355  !contain(toExcludeFrom, header_name)))
356  {
357  std::string raw_txt = raw_comment->getRawText(ast_ctxt.getSourceManager()).str();
358  raw_text.erase(raw_text.find("*/")+2,std::string::npos);
359  unsigned raw_text_len = raw_text.length();
360 
361  SourceManager &src_mgr = ast_ctxt.getSourceManager();
362  FileManager &file_mgr = src_mgr.getFileManager();
363  FileID fid = src_mgr.getMainFileID();
364 
365  const FileEntry *hdr_fentry = nullptr;
366  for (auto it = headersDirectories.begin();
367  it != headersDirectories.end();
368  ++it)
369  {
370  std::string hdr_dir = (*it).append("/");
371  hdr_fentry = file_mgr.getFile(hdr_dir.append(header_name), true);
372  if (hdr_fentry)
373  break;
374  }
375  // Found hdr file included here
376  if (hdr_fentry)
377  {
378  auto buf = src_mgr.getMemoryBufferForFile(hdr_fentry);
379  const char *pbuf = buf->getBufferStart();
380  std::string pbufstr = std::string(pbuf);
381 
382  unsigned int hdr_lines = count_buffer_chars_number(pbufstr, "\n");
383  // Get location of start of included hdr file
384  unsigned comment_start_file_offset = src_mgr.getFileOffset(comment_start_loc);
385  unsigned comment_end_file_offset = comment_start_file_offset + raw_text_len -1;
386  unsigned comment_end_line_number = src_mgr.getLineNumber(fid, comment_end_file_offset+1);
387  unsigned hdr_start_line_number = comment_end_line_number +1;
388  unsigned hdr_end_line_number = hdr_start_line_number + hdr_lines;
389  // outs() << "Hdr start line number = " << hdr_start_line_number << "\n";
390  // outs() << "Hdr end line number = " << hdr_end_line_number << "\n";
391  SourceLocation hdr_start = src_mgr.translateLineCol(fid, hdr_start_line_number, 1);
392  SourceLocation hdr_end = src_mgr.translateLineCol(fid, hdr_end_line_number, 1);
393 
394  emitDiagAndFix(hdr_start, hdr_end,
395  header_name);
396  }
397  }
398  }

424  std::vector<std::string>
425  DeIncludePreProC::tokenizeString(const std::string& str,
426  const std::string& delims)
427  {
428  std::vector<std::string> tokens;
429 
430  std::string::size_type lastPos = str.find_first_not_of(delims, 0);
431  std::string::size_type pos = str.find_first_of(delims, lastPos);
432 
433  while (pos != std::string::npos ||
434  lastPos != std::string::npos)
435  {
436  tokens.push_back(str.substr(lastPos, pos - lastPos));
437  lastPos = str.find_first_not_of(delims, pos);
438  pos = str.find_first_of(delims, lastPos);
439  }
440  return tokens;
441  }
