409  const std::string& cxxRecordName,
410  const std::string& memberName)
411  {
412 #ifdef ACTIVATE_TRACES
413  outs() << "findCXXRecordMemberInTranslationUnit searching '" << cxxRecordName << "' '" << memberName << "'\n";
414 #endif // !ACTIVATE_TRACES

416  // Let's first search is root context
417  DeclContext *declCtxt = TranslationUnitDecl::castToDeclContext(transUnit);
418  SmallVector<DeclContext *, 10> declCtxts;
419  // Then in all semantically connected contexts (ex. namespaces)
420  const_cast<TranslationUnitDecl *>(transUnit)->collectAllContexts(declCtxts);
421  // Add root context in vector for having one loop only
422  // There is better chances to find in connected context instead of root one, then add root at end
423  declCtxts.push_back(declCtxt);
424 
425  for (auto ctxtsit = declCtxts.begin(); ctxtsit != declCtxts.end(); ++ctxtsit)
426  {
427  DeclContext *aDeclCtxt = *ctxtsit;
428  for (auto declit = aDeclCtxt->decls_begin(); declit != aDeclCtxt->decls_end(); ++declit)
429  {
430  std::string recordName;
431  const Decl *aDecl = *declit;
432  if (isa<NamedDecl>(*aDecl))
433  {
434  const NamedDecl *namedDecl = dyn_cast<NamedDecl>(aDecl);
435  recordName = namedDecl->getNameAsString();
436  }
437  if (isa<CXXRecordDecl>(*aDecl) &&
438  !recordName.empty() &&
439  (cxxRecordName.find(recordName) != std::string::npos ||
440  std::string("struct ").append(recordName).compare(cxxRecordName) == 0))
441  {
442  const RecordDecl *recordDecl = dyn_cast<RecordDecl>(aDecl);
443  // iterate on record member fields
444  for (auto fieldsit = recordDecl->field_begin(); fieldsit != recordDecl->field_end(); ++fieldsit)

457  QualType qtype = valueDecl->getType();

459  fieldTypeName = QualType::getAsString(qt_split);

464  {
465  const ConstantArrayType *catype = fieldDecl->getASTContext().getAsConstantArrayType(qtype);
466  ret["elementType"] = catype->getElementType().getAsString();
467  ret["elementSize"] = catype->getSize().toString(10, false);
468  }
469

554 #ifdef ACTIVATE_TRACES

557  std::cout << "matches[" << ni << "] = " << matches[ni].str() << "\n";
558 #endif // !ACTIVATE_TRACES
559  if (! indicator)
560  {
561  size_t derefpos;
562  // First trim out all unwanted spaces in case there is some
563  var["full"] = matches[PAGESJAUNES_REGEX_HOSTVAR_DECODE_RE_FULLMATCH].trim(" \n\t");
564  var["hostvar"] = matches[PAGESJAUNES_REGEX_HOSTVAR_DECODE_RE_HOSTVAR].trim(" \n\t");
565  var["hostmember"] = matches[PAGESJAUNES_REGEX_HOSTVAR_DECODE_RE_HOSTMEMBER].trim(" \n\t");

569  var["deref"] = ".";
570  if (!var["deref"].empty())
571  var["hostrecord"] = var["hostvar"].substr(0, derefpos);
572  else
573  var["hostrecord"] = var["hostmember"];
574  Regex trimIdentifierRe(PAGESJAUNES_REGEX_TRIM_IDENTIFIER_RE);
575  SmallVector<StringRef, 8> idmatches;
576  if (trimIdentifierRe.match(var["hostrecord"], &idmatches))
577  var["hostrecord"] = idmatches[PAGESJAUNES_REGEX_TRIM_IDENTIFIER_RE_IDENTIFIER].trim(" \n\t");
578  }
579  else
580  {
581  size_t derefipos;
582  var["fulli"] = matches[PAGESJAUNES_REGEX_HOSTVAR_DECODE_RE_FULLMATCH].trim(" \n\t");

586  var["derefi"] = "->";
587  else if ((derefipos = var["hostvari"].find(".")) != std::string::npos)
588  var["derefi"] = ".";

594  var["hostrecordi"] = var["hostvari"].substr(0, derefipos);

598  SmallVector<StringRef, 8> idmatches;
599  if (trimIdentifierRe.match(var["hostrecordi"], &idmatches))

601  }
602 
603 #ifdef ACTIVATE_RESULT_TRACES

666  createBackupFile(const std::string& pathname)
667  {
668  struct stat buffer;
669 
670  // Open the file to backup
671  std::ifstream src(pathname.c_str(), std::ios::binary);
672  if (src.is_open())
673  {
674  std::string backupPathname(pathname.c_str());
675  backupPathname.append(".bak");
676 
677  // First find if this bak is free (do not already exist)
678  bool do_exist = true;
679  int baknum = 0;
680  do
681  if ((do_exist = (stat (backupPathname.c_str(), &buffer)) == 0))
682  {
683  backupPathname.assign(pathname);
684  int nbaknum = (int)(floor(log10(baknum)));
685 #ifdef ACTIVATE_TRACES
686  llvm::outs() << "baknum = " << baknum << "\n";
687  llvm::outs() << "nbaknum = " << nbaknum << "\n";
688  llvm::outs() << "floor(log10(baknum)) = " << nbaknum << "\n";
689 #endif // !ACTIVATE_TRACES
690  if (baknum == 0)
691  backupPathname.append("-0.bak");
692 
693  else
694  {
695  int baknum_temp = baknum;
696  char digit = '0';
697  backupPathname.append("-");
698  for (int n = nbaknum; n >= 0; n--)
699  {
700 #ifdef ACTIVATE_TRACES
701  llvm::outs() << "n = " << n << "\n";
702 #endif // !ACTIVATE_TRACES
703  digit = (char)(floor(baknum_temp / exp10(n))) + '0';
704  baknum_temp -= (floor(baknum_temp / exp10(n)))*exp10(n);
705 #ifdef ACTIVATE_TRACES
706  llvm::outs() << "digit = floor(baknum / exp10(n)) = " << digit << "\n";
707  llvm::outs() << "baknum_temp = " << baknum_temp << "\n";
708 #endif // !ACTIVATE_TRACES
709  backupPathname.append(1, digit);
710  }
711  backupPathname.append(".bak");
712  }
713 #ifdef ACTIVATE_TRACES
714  llvm::outs() << "trying new bak path = " << backupPathname << "\n";
715 #endif // !ACTIVATE_TRACES
716  baknum++;
717  }
718  while (do_exist);
719 
720 #ifdef ACTIVATE_TRACES
721  llvm::outs() << "found free bak path = " << backupPathname << "\n";
722 #endif // !ACTIVATE_TRACES
723  // Ok we got a free backup file name, then let's go
724  std::ofstream dst(backupPathname.c_str(), std::ios::binary);
725  if (dst.is_open())
726  {
727  dst << src.rdbuf();
728  dst.close();
729  }
730  src.close();
731  }
