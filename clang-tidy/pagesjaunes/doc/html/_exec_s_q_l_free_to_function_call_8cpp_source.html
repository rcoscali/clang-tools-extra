349  bool
350  ExecSQLFreeToFunctionCall::processTemplate(const std::string& tmpl,
351  const std::string& fname,
352  string2_map& values_map)
353  {
354  // Return value
355  bool ret = false;
356  // File buffers used for I/O
357  std::filebuf fbi, fbo;
358 
359  do
360  {
361  // Open template
362  if (!fbi.open(tmpl, std::ios::in))
363  break;
364 
365  // Open output file
366  if (!fbo.open(fname, std::ios::out))
367  break;
368 
369  // Input and output streams
370  std::istream is(&fbi);
371  std::ostream os(&fbo);
372 
373  // While not end of template
374  while (is)
375  {
376  // Buffer for input line
377  char buf[1024];
378  // Read line in buffer
379  is.getline(buf, 255);
380  std::string aline = buf;
381  // Iterate on all map values
382  for (auto it = values_map.begin();
383  it != values_map.end();
384  ++it)
385  {
386  // Position of found value name
387  size_t pos = 0;
388  do
389  {
390  // If we found the current value name
391  if ((pos = aline.find(it->first, pos)) != std::string::npos)
392  {
393  // Erase value name at found position in input line
394  aline.erase(pos, it->first.length());
395  // Insert value replacement at the same position
396  aline.insert(pos, it->second);
397  }
398  }
399  // Do it until end of line
400  while (pos != std::string::npos);
401  }
402 
403  // Output processed line
404  os << aline << std::endl;
405  }
406 
407  // All was ok
408  ret = true;
409  }
410  while (0);
411 
412  // Close output file
413  if (fbo.is_open())
414  fbo.close();
415  // Close input file
416  if (fbi.is_open())
417  fbi.close();
418 
419  // Return status
420  return ret;
421  }

621  void
622  ExecSQLFreeToFunctionCall::check(const MatchFinder::MatchResult &result)
623  {
624  map_replacement_values rv;
625 
626  // Get the source manager
627  SourceManager &srcMgr = result.Context->getSourceManager();
628  DiagnosticsEngine &diagEngine = result.Context->getDiagnostics();
629 
630  /*
631  * Init check context
632  * ------------------
633  */
634 
635  // Get the compound statement AST node as the bounded var 'proCBlock'
636  const CompoundStmt *stmt = result.Nodes.getNodeAs<CompoundStmt>("proCBlock");
637 
638  // Get start/end locations for the statement
639  SourceLocation loc_start = stmt->getLocStart();
640  SourceLocation loc_end = stmt->getLocEnd();
641 
642  // Get file ID in src mgr
643  FileID startFid = srcMgr.getFileID(loc_start);
644  // Get compound statement start line num
645  unsigned startLineNum = srcMgr.getLineNumber(startFid, srcMgr.getFileOffset(loc_start));
646  std::stringbuf slnbuffer;
647  std::ostream slnos (&slnbuffer);
648  slnos << startLineNum;
649  std::string originalSourceFilename = srcMgr.getFileEntryForID(srcMgr.getMainFileID())->getName().str().append(slnbuffer.str().insert(0, "#"));
650 
651  //outs() << "Found one result at line " << startLineNum << " of file '" << originalSourceFilename << "\n";
652 
653  /*
654  * Find the comment for the EXEC SQL statement
655  * -------------------------------------------
656  *
657  * Iterate from line -2 to line -5 until finding the whole EXEC SQL comment
658  * comments are MAX_NUMBER_OF_LINE_TO_SEARCH lines max
659  */
660 
661  // Current line number
662  size_t lineNum = startLineNum-2;
663 
664  // Get end comment loc
665  SourceLocation comment_loc_end = srcMgr.translateLineCol(startFid, lineNum, 1);
666  SourceLocation comment_loc_start;
667 
668  // Does error occured while doing getCharacterData
669  bool errOccured = false;
670 
671  // Get comment C string from the file memory buffer
672  const char *commentCStr = srcMgr.getCharacterData(comment_loc_end, &errOccured);
673 
674  // The line data & comment
675  std::string lineData;
676 
677  // Offset of comment start in file
678  size_t commentStart = 0;
679 
680  // If #line is available keep line number and filename of the .pc file
681  unsigned int pcLineNum = 0;
682  std::string pcFilename;
683  bool found_line_info = false;
684 
685  /*
686  * Finding loop
687  * ------------
688  */
689  do
690  {
691  // All was ok ?
692  if (!errOccured)
693  {
694  // Assign comment C string to line data var
695  lineData.assign(commentCStr);
696  // And remove all remaining chars after end of line
697  lineData.erase (lineData.find("\n", 1), std::string::npos);
698 
699  // Try to get comment start in current line
700  commentStart = lineData.find("/*");
701 
702  if (lineData.find("#line ") != std::string::npos)
703  {
704  // Keep the line number and file:
705  // Format of the preprocessor #line
706  // #line <linenum> <filepath>
707  // linenum is an unsigned int
708  // and filepath is a dbl-quoted string
709  Regex lineDefineRe("^#line ([0-9]+) \"(.*)\"$");
710  SmallVector<StringRef, 8> lineMatches;
711  if (lineDefineRe.match(lineData, &lineMatches))
712  {
713  found_line_info = true;
714  std::istringstream isstr;
715  std::stringbuf *pbuf = isstr.rdbuf();
716  pbuf->str(lineMatches[1]);
717  isstr >> pcLineNum;
718  pcFilename = lineMatches[2];
719  }
720  else
721  {
722  errs() << "Cannot match a #line definition !\n";
723  }
724  }
725 
726  // If found, break the loop
727  if (commentStart != std::string::npos)
728  break;
729  }
730 
731  else
732  // Error occured at getting char data ! break comment start find loop
733  break;
734 
735  // try previous line
736  lineNum--;
737  // Get Location for the new current line
738  comment_loc_start = srcMgr.translateLineCol(startFid, lineNum, 1);
739  // And get char data again
740  commentCStr = srcMgr.getCharacterData(comment_loc_start, &errOccured);
741  }
742  // Try again
743  while (lineNum > 0);
744 
745  /*
746  * Comment found
747  */
748  std::string comment;
749  std::string function_name;
750 
751  // If found comment start & no error occured
752  if (lineNum > 0 && !errOccured)
753  {
754  // Assign comment char data in the comnt string
755  comment = std::string(commentCStr);
756 
757  /*
758  * Erase from end of comment to end of data
759  */
760 
761  // Find end of comment
762  size_t endCommentPos = comment.find("*/", 1);
763 
764  // Erase until end of character data
765  comment.erase (endCommentPos +2, std::string::npos);
766 
767  /*
768  * Remove all \n from comment
769  */
770 
771  // Find CR from start in comment
772  size_t crpos = comment.find("\n", 0);
773  bool had_cr = false;
774 
775  // Iterate
776  do
777  {
778  // If one has been found
779  if (crpos != std::string::npos)
780  {
781  // Erase CR
782  comment.erase(crpos, 1);
783  had_cr = true;
784  }
785  // Find again CR
786  crpos = comment.find("\n", 0);
787  }
788  // Until no more is found
789  while (crpos != std::string::npos);
790 
791  //outs() << "comment for compound statement at line #" << startLineNum << ": '" << comment << "'\n";
792 
793  /*
794  * Create function call for the request
795  */
796 
797  // Regex for processing comment for all remaining requests
798  Regex freeReqRe(PAGESJAUNES_REGEX_EXEC_SQL_FREE_REQ_RE);
799 
800  // Returned matches
801  SmallVector<StringRef, 8> matches;
802 
803  /*
804  * Now we match against a more permissive regex for all other (simpler) requests
805  * =============================================================================
806  */
807  if (freeReqRe.match(comment, &matches))
808  {
809  std::string reqAllocName = matches[2];
810  std::string requestExecSql = "FREE :";
811  requestExecSql.append(reqAllocName);
812 
813  // Append first match (action) to function name
814  function_name.append("free");
815 
816  // Get match in rest string
817  std::string rest(reqAllocName);
818  // Capitalize it
819  rest[0] &= ~0x20;
820  // And append to function name
821  function_name.append(rest);
822 
823  if (generation_do_report_modification_in_pc)
824  {
825  std::ostringstream had_cr_strstream;
826  had_cr_strstream << had_cr;
827  rv.insert(std::pair<std::string, std::string>("had_cr", had_cr_strstream.str()));
828  rv.insert(std::pair<std::string, std::string>("fullcomment", comment));
829  rv.insert(std::pair<std::string, std::string>("reqname", reqAllocName));
830  rv.insert(std::pair<std::string, std::string>("funcname", function_name));
831  rv.insert(std::pair<std::string, std::string>("execsql", requestExecSql));
832  if (found_line_info)
833  {
834  std::ostringstream pc_line_num;
