229  void
230  BackupFile::SHA256::transform(const unsigned char *message, unsigned int block_nb)
231  {
232  uint32 w[64];
233  uint32 wv[8];
234  uint32 t1, t2;
235  const unsigned char *sub_block;
236  int i;
237  int j;
238  for (i = 0; i < (int) block_nb; i++) {
239  sub_block = message + (i << 6);
240  for (j = 0; j < 16; j++) {
241  SHA2_PACK32(&sub_block[j << 2], &w[j]);
242  }
243  for (j = 16; j < 64; j++) {
244  w[j] = SHA256_F4(w[j - 2]) + w[j - 7] + SHA256_F3(w[j - 15]) + w[j - 16];
245  }
246  for (j = 0; j < 8; j++) {
247  wv[j] = m_h[j];
248  }
249  for (j = 0; j < 64; j++) {
250  t1 = wv[7] + SHA256_F2(wv[4]) + SHA2_CH(wv[4], wv[5], wv[6])
251  + sha256_k[j] + w[j];
252  t2 = SHA256_F1(wv[0]) + SHA2_MAJ(wv[0], wv[1], wv[2]);
253  wv[7] = wv[6];
254  wv[6] = wv[5];
255  wv[5] = wv[4];
256  wv[4] = wv[3] + t1;
257  wv[3] = wv[2];
258  wv[2] = wv[1];
259  wv[1] = wv[0];
260  wv[0] = t1 + t2;
261  }
262  for (j = 0; j < 8; j++) {
263  m_h[j] += wv[j];
264  }
265  }
266  }

283  void
284  BackupFile::SHA256::update(const unsigned char *message, unsigned int len)
285  {
286  unsigned int block_nb;
287  unsigned int new_len, rem_len, tmp_len;
288  const unsigned char *shifted_message;
289  tmp_len = BackupFile::SHA256::SHA224_256_BLOCK_SIZE - m_len;
290  rem_len = len < tmp_len ? len : tmp_len;
291  memcpy(&m_block[m_len], message, rem_len);
292  if (m_len + len < BackupFile::SHA256::SHA224_256_BLOCK_SIZE) {
293  m_len += len;
294  return;
295  }
296  new_len = len - rem_len;
297  block_nb = new_len / BackupFile::SHA256::SHA224_256_BLOCK_SIZE;
298  shifted_message = message + rem_len;
299  transform(m_block, 1);
300  transform(shifted_message, block_nb);
301  rem_len = new_len % BackupFile::SHA256::SHA224_256_BLOCK_SIZE;
302  memcpy(m_block, &shifted_message[block_nb << 6], rem_len);
303  m_len = rem_len;
304  m_tot_len += (block_nb + 1) << 6;
305  }

307  void
308  BackupFile::SHA256::final(unsigned char *digest)
309  {
310  unsigned int block_nb;
311  unsigned int pm_len;
312  unsigned int len_b;
313  int i;
314 
315  block_nb = (1 + ((BackupFile::SHA256::SHA224_256_BLOCK_SIZE - 9)
316  < (m_len % BackupFile::SHA256::SHA224_256_BLOCK_SIZE)));
317  len_b = (m_tot_len + m_len) << 3;
318  pm_len = block_nb << 6;
319  memset(m_block + m_len, 0, pm_len - m_len);
320  m_block[m_len] = 0x80;
321  SHA2_UNPACK32(len_b, m_block + pm_len - 4);
322  transform(m_block, block_nb);
323  for (i = 0 ; i < 8; i++)
324  SHA2_UNPACK32(m_h[i], &digest[i << 2]);
325  }

327  std::string
328  BackupFile::sha256(const std::string& input)
329  {
330  unsigned char digest[BackupFile::SHA256::DIGEST_SIZE];
331  memset(digest, 0, BackupFile::SHA256::DIGEST_SIZE);
332 
333  BackupFile::SHA256 ctx = BackupFile::SHA256();
334  ctx.init();
335  ctx.update(reinterpret_cast<const unsigned char *>(input.c_str()), input.length());
336  ctx.final(digest);
337 
338  char buf[2 * BackupFile::SHA256::DIGEST_SIZE+1];
339  buf[2 * BackupFile::SHA256::DIGEST_SIZE] = 0;
340  for (unsigned int i = 0; i < BackupFile::SHA256::DIGEST_SIZE; i++)
341  sprintf(buf+i*2, "%02x", digest[i]);
342  return std::string(buf);
343  }
